import requests

url = "http://service.netsage.click:3738/api/v1/plgi/search"
payload = {
    "url": "https://example.com",
    "fetch_engine": "pl_chrome",
    "parser": "sum_v2",
    "key": "your-api-key"
}

response = requests.post(url, json=payload)
data = response.json()

print(data['data']['results']['response']['md'])

cURL示例 - 网络检索 
import requests

url = "http://service.netsage.click:3738/api/v1/plgi/search"
payload = {
    "query": "如何培养良好的睡眠习惯",
    "count": 3,
    "search_engine": "baidu",
    "key": "your-api-key"
}

response = requests.post(url, json=payload)
data = response.json()

# 打印搜索结果
for i, result in enumerate(data['data']['results'], 1):
    print(f"{i}. {result['siteName']}")
    print(f"   {result['snippet'][:100]}...")
    print(f"   URL: {result['url']}\n")
